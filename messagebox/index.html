            loadingbox( "提交中...", "立刻跳转" ,function(obj){
                setTimeout(function(){
                    var n = 3,
                    timer = setInterval(function(){
                        n --;
                        obj.text(n + "秒后自动跳转");
                        if(n == 0){
                            clearInterval(timer);
                            obj.hide();
                        }	
                    },1000);
                    obj.broken("3秒后自动跳转",function(){
                        //点击事件
                        clearInterval(timer);
                        obj.text("点击");	
                    });
                },1600);
            });
