loadingbox( [提示文案string] , [按钮文案string] , [回调函数function] ) 
            loadingbox("提交中...",function(obj){
                setTimeout(function(){
                    obj.hide();//关闭
                },1600);
            });
            
            loadingbox("提交中...",function(obj){
                setTimeout(function(){
                    obj.broken("结束loading，并设定提示文案和按钮的点击绑定事件",function(){
                        messagebox("点击");
                    });
                },1600);
            });
            
            loadingbox( "提交中...", "立刻跳转" ,function(obj){
                setTimeout(function(){
                    var n = 3,
                    timer = setInterval(function(){
                        n --;
                        obj.text(n + "秒后自动跳转");
                        if(n == 0){
                            clearInterval(timer);
                            obj.hide();
                        }	
                    },1000);
                    obj.broken("3秒后自动跳转",function(){
                        //点击事件
                        clearInterval(timer);
                        obj.text("点击了");
                        setTimeout(obj.hide,1000);		
                    });
                },1600);
            });
