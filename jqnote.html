全局设置ajax loading动画以及错误回调 
            	$(document).ajaxStart(function() {
               	    $("#loading").show();
             	});
                $(document).ajaxStop(function() {
               	    $("#loading").hide();
             	});
                $(document).ajaxError(function(event, request, settings, thrownError) {
                    swal("提示", "系统错误！", "error")
             	});
